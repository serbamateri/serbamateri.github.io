*

putih = 200

*

a = jumlah yang terjual

*

120 - a = bm

200 - a = bp

u₆= 3. r⁵ = 96

r⁵ = 32

r = 2

u₂= a.r = 3. 2 = 6 cm

u₃= u₂ . r = 6 .2 = 12 cm

u₄= u₃ . r = 12.2 = 24 cm

u₅ =u₄ . r. = 24.2 = 48 cm
