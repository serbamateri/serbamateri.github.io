cos a + cos b = ... (lihat di buku ada)

[tex]=2cos\left(\frac{130^o+110^o}{2}\right)cos\left(\frac{130^o-110^o}{2}\right)+cos10^o[/tex]

[tex]=2cos120^ocos10^o+cos10^o[/tex]
[tex]=2*(\frac{-1}{2})cos10^o+cos10^o[/tex]
[tex]=-cos10^o+cos10^o[/tex]
[tex]=0[/tex]
